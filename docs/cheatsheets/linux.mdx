## üìÅ File & Directory Navigation

```bash
# Show current directory
pwd

# List contents
ls        # basic
ls -l     # long format
ls -a     # include hidden files
ls -lh    # human-readable

# Change directory

cd ..     # up one level

# Make or remove directories
mkdir newfolder
mkdir -p a/b/c      # nested dirs
rmdir emptyfolder
rm -r folder        # non-empty
```

## üîç Advanced File Operations

### Text Processing

```bash
# Sort and filter
sort file.txt
sort -r file.txt        # reverse
sort -n numbers.txt     # numeric
uniq file.txt           # remove duplicates
uniq -c file.txt        # count occurrences

# Cut and paste
cut -d',' -f1,3 file.csv    # extract columns
paste file1.txt file2.txt   # merge files

# Advanced grep
grep -i "pattern" file.txt     # case insensitive
grep -v "pattern" file.txt     # invert match
grep -n "pattern" file.txt     # show line numbers
grep -A 3 -B 3 "pattern" file.txt  # context lines
grep -r --include="*.js" "function" .  # recursive with file filter

# sed (stream editor)
sed 's/old/new/g' file.txt     # replace text
sed -i 's/old/new/g' file.txt  # edit in place
sed '1,5d' file.txt            # delete lines 1-5
sed -n '10,20p' file.txt       # print lines 10-20

# awk (pattern scanning)
awk '{print $1}' file.txt      # print first column
awk -F',' '{print $2}' file.csv # comma-separated
awk '/pattern/ {print $0}' file.txt  # pattern matching
```

### Advanced Find Operations

```bash
# Find files by various criteria
find . -name "*.js" -type f
find . -size +100M              # files larger than 100MB
find . -mtime -7                # modified in last 7 days
find . -user username           # owned by user
find . -perm 755                # specific permissions

# Find and execute commands
find . -name "*.log" -exec rm {} \;
find . -name "*.tmp" -delete
find . -type d -empty           # empty directories

# Advanced find with multiple conditions
find . \( -name "*.js" -o -name "*.ts" \) -size +1k
```

## üîí Security & Permissions

### Advanced Permissions

```bash
# Octal permissions
chmod 755 file                  # rwxr-xr-x
chmod 644 file                  # rw-r--r--
chmod 600 file                  # rw-------
chmod +x script.sh              # add execute

# Special permissions
chmod +s file                   # setuid
chmod +t directory              # sticky bit
chmod 1777 /tmp                 # sticky bit example

# ACL (Access Control Lists)
setfacl -m u:username:rwx file
getfacl file
setfacl -x u:username file      # remove ACL
```

### SSH & Remote Access

```bash
# SSH operations
ssh user@hostname
ssh -p 2222 user@hostname       # custom port
ssh -i keyfile user@hostname    # key file
ssh-copy-id user@hostname       # copy public key

# SSH config
# ~/.ssh/config
Host myserver
    HostName server.com
    User myuser
    Port 2222
    IdentityFile ~/.ssh/id_rsa

# SSH tunnels
ssh -L 8080:localhost:80 user@server    # local port forward
ssh -R 8080:localhost:80 user@server    # remote port forward
```

## üêö Shell Scripting

### Basic Scripting

```bash
#!/bin/bash
# Shebang line

# Variables
NAME="John"
AGE=25
echo "Hello $NAME, you are $AGE years old"

# Command substitution
DATE=$(date)
echo "Current date: $DATE"

# Conditional statements
if [ $AGE -gt 18 ]; then
    echo "Adult"
else
    echo "Minor"
fi

# Loops
for i in {1..5}; do
    echo "Number: $i"
done

# Functions
greet() {
    echo "Hello $1"
}
greet "World"
```

### Advanced Scripting

```bash
# Error handling
set -e                          # exit on error
set -u                          # exit on undefined variable
set -o pipefail                 # exit on pipe failure

# Input validation
if [ $# -eq 0 ]; then
    echo "Usage: $0 <filename>"
    exit 1
fi

# Logging
log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" >> /var/log/script.log
}

# Signal handling
trap 'echo "Script interrupted"; exit 1' INT TERM
```

## üîß Advanced Networking

### Network Troubleshooting

```bash
# Connectivity tests
ping -c 4 google.com            # ping 4 times
traceroute google.com           # trace route
mtr google.com                  # continuous traceroute

# Port scanning
nmap -p 80,443 hostname         # scan specific ports
nmap -sS hostname               # stealth scan
nmap -A hostname                # aggressive scan

# Network capture
tcpdump -i eth0 port 80         # capture HTTP traffic
tcpdump -i any -n host 192.168.1.1  # capture specific host
```

### Firewall Management

```bash
# UFW (Ubuntu)
sudo ufw enable
sudo ufw allow 22               # allow SSH
sudo ufw allow from 192.168.1.0/24
sudo ufw deny 80
sudo ufw status

# iptables
sudo iptables -L                # list rules
sudo iptables -A INPUT -p tcp --dport 22 -j ACCEPT
sudo iptables -A INPUT -j DROP
```

## üß™ Advanced Commands

### Compression & Archives

```bash
# tar operations
tar -czf archive.tar.gz files/  # create gzip archive
tar -xzf archive.tar.gz         # extract gzip archive
tar -cjf archive.tar.bz2 files/ # create bzip2 archive
tar -tf archive.tar.gz          # list contents

# zip operations
zip -r archive.zip files/       # create zip
unzip archive.zip               # extract zip
unzip -l archive.zip            # list contents

# 7zip operations
7z a archive.7z files/          # create 7z
7z x archive.7z                 # extract 7z
```

### Environment & Aliases

```bash
# Environment variables
export PATH=$PATH:/new/path
export EDITOR=vim
env                             # show all variables
printenv VARIABLE               # show specific variable

# Aliases
alias ll='ls -alF'
alias la='ls -A'
alias l='ls -CF'
alias grep='grep --color=auto'
alias ..='cd ..'
alias ...='cd ../..'

# Functions
mkcd() { mkdir -p "$1" && cd "$1"; }
extract() {
    if [ -f $1 ] ; then
        case $1 in
            *.tar.bz2)   tar xjf $1     ;;
            *.tar.gz)    tar xzf $1     ;;
            *.bz2)       bunzip2 $1     ;;
            *.rar)       unrar e $1     ;;
            *.gz)        gunzip $1      ;;
            *.tar)       tar xf $1      ;;
            *.tbz2)      tar xjf $1     ;;
            *.tgz)       tar xzf $1     ;;
            *.zip)       unzip $1       ;;
            *.Z)         uncompress $1  ;;
            *.7z)        7z x $1        ;;
            *)           echo "'$1' cannot be extracted via extract()" ;;
        esac
    else
        echo "'$1' is not a valid file"
    fi
}
```

### Job Control

```bash
# Background jobs
command &                       # run in background
jobs                            # list jobs
fg %1                          # bring job 1 to foreground
bg %1                          # send job 1 to background
nohup command &                 # run immune to hangups

# Process substitution
diff <(command1) <(command2)    # compare outputs
comm <(sort file1) <(sort file2) # compare sorted files
```
